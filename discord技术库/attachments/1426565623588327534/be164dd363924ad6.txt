<script>
let f=!1,a=document.querySelector('.front-logo'),b=document.querySelector('.back-logo');
document.getElementById('flipCard').onclick=()=>{
  f=!f;a.style.animation='';b.style.animation='';
  if(f){
    a.style.animation='flip360 0.8s forwards';
    setTimeout(()=>{a.style.display='none';b.style.display='block';b.style.animation='showWithFade 0.8s forwards'},800)
  }else{
    b.style.animation='flip360 0.8s forwards';
    setTimeout(()=>{b.style.display='none';a.style.display='block';a.style.animation='showWithFade 0.8s forwards'},800)
  }
};
</script>
