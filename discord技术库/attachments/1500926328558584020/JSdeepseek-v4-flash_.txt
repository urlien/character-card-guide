<!-- 利用 onerror 事件注入 JS，获取聊天滚动容器并暴露全局滚动方法 -->
<img src="x" style="position:absolute; width:0; height:0; display:none;" onerror="(function(){
    try {
        // 查找所有 .uni-scroll-view 容器（通常有多个，第二个即聊天区域）
        var allScrollView = document.querySelectorAll('.uni-scroll-view');
        var chatBox = allScrollView.length > 1 ? allScrollView[1] : (allScrollView.length > 0 ? allScrollView[0] : null);
        if (!chatBox) return;
        
        // 全局滚动到顶部
        window.scrollToTop = function() {
            chatBox.scrollTo({ top: 0, behavior: 'smooth' });
        };
        
        // 全局滚动到底部
        window.scrollToBottom = function() {
            chatBox.scrollTo({ top: chatBox.scrollHeight, behavior: 'smooth' });
        };

        // 全局字体缩放函数（放大）
        window.zoomIn = function() {
            var scale = window.tpZoomScale || 1;
            scale = Math.min(scale * 1.05, 3); // 最大放大3倍
            window.tpZoomScale = scale;
            var els = document.querySelectorAll('.chat-body .content.left, .chat-body .content.left *, .chat-body .content.right, .chat-body .content.right *');
            for (var i = 0; i < els.length; i++) {
                var el = els[i];
                if (!el.getAttribute('data-tp-fs')) {
                    el.setAttribute('data-tp-fs', window.getComputedStyle(el).fontSize);
                }
                el.style.fontSize = (parseFloat(el.getAttribute('data-tp-fs')) * scale) + 'px';
            }
        };

        // 全局字体缩放函数（缩小）
        window.zoomOut = function() {
            var scale = window.tpZoomScale || 1;
            scale = Math.max(scale * 0.95, 0.3); // 最小缩小到30%
            window.tpZoomScale = scale;
            var els = document.querySelectorAll('.chat-body .content.left, .chat-body .content.left *, .chat-body .content.right, .chat-body .content.right *');
            for (var i = 0; i < els.length; i++) {
                var el = els[i];
                if (!el.getAttribute('data-tp-fs')) {
                    el.setAttribute('data-tp-fs', window.getComputedStyle(el).fontSize);
                }
                el.style.fontSize = (parseFloat(el.getAttribute('data-tp-fs')) * scale) + 'px';
            }
        };
    } catch(e) {}
    // 执行完毕后移除自身，避免残留
    this.remove();
}).call(this)">

    <div id="hu-theme-btn" 
         onclick="event.stopPropagation();
                 var sw = document.getElementById('hu-style-warm');
                 var sd = document.getElementById('hu-style-dark');
                 var s = parseInt(this.getAttribute('data-s') || '0');
                 s = (s + 1) % 3;
                 this.setAttribute('data-s', s);
                 if (s == 0) {
                     // 启用暖杏主题，禁用暗黑
                     if (sw) sw.disabled = false;
                     if (sd) sd.disabled = true;
                     this.style.backgroundColor = '#D4A5A5';
                     this.style.borderColor = '#594D46';
                     this.style.color = '#FFFCFA';
                     this.innerText = '暖杏';
                 } else if (s == 1) {
                     // 启用暗黑主题，禁用暖杏
                     if (sw) sw.disabled = true;
                     if (sd) sd.disabled = false;
                     this.style.backgroundColor = '#8a0303';
                     this.style.borderColor = '#ff4d4d';
                     this.style.color = '#EAEAEA';
                     this.innerText = '暗黑';
                 } else {
                     // 关闭所有主题样式
                     if (sw) sw.disabled = true;
                     if (sd) sd.disabled = true;
                     this.style.backgroundColor = '#333333';
                     this.style.borderColor = '#1a1a1a';
                     this.style.color = '#999999';
                     this.innerText = '关闭';
                 }"
         data-s="0" 
         style="box-sizing:border-box; width:26px; padding:15px 2px; background-color:#D4A5A5; border:1px solid #594D46; border-left:none; border-radius:0 8px 8px 0; color:#FFFCFA; cursor:pointer; writing-mode:vertical-rl; text-orientation:mixed; font-size:14px; letter-spacing:3px; text-align:center; transition:all 0.3s cubic-bezier(0.4,0,0.2,1); box-shadow:2px 2px 6px rgba(0,0,0,0.3);">
         暖杏
    </div>
