    <script>
        // 核心时间管理系统
        let currentDateTime = new Date();
        // 时间格式化 HH:mm
        function formatTime(date) {
            const hours = String(date.getHours()).padStart(2, '0');
            const minutes = String(date.getMinutes()).padStart(2, '0');
            return `${hours}:${minutes}`;
        }
        // 日期格式化 MM月DD日
        function formatDate(date) {
            const month = String(date.getMonth() + 1).padStart(2, '0');
            const day = String(date.getDate()).padStart(2, '0');
            return `${month}月${day}日`;
        }
        // 时间自动更新逻辑，保留原有的23点自动跳次日规则
        function updateDateTime() {
            currentDateTime.setMinutes(currentDateTime.getMinutes() + 30);
            // 晚23点后自动跳转至第二天00:00
            if (currentDateTime.getHours() >= 23) {
                currentDateTime.setDate(currentDateTime.getDate() + 1);
                currentDateTime.setHours(0, 0, 0, 0);
            }
            updateDateTimeDisplay();
        }
        // 更新页面时间日期显示
        function updateDateTimeDisplay() {
            document.getElementById('currentTime').textContent = formatTime(currentDateTime);
            document.getElementById('currentDate').textContent = formatDate(currentDateTime);
        }
        // 进度条统一更新函数
        function updateProgress(barId, percentId, percent) {
            const bar = document.getElementById(barId);
            const percentText = document.getElementById(percentId);
            bar.style.width = `${percent}%`;
            if (percentText) percentText.textContent = `${percent}%`;
        }
        // 页面初始化
        window.onload = function() {
            // 初始化时间显示
            updateDateTimeDisplay();
            // 初始化所有进度条
            updateProgress('gatherProgress', null, 0);
            updateProgress('huntProgress', null, 0);
            updateProgress('breedProgress', null, 0);
            updateProgress('orgasmProgress', 'orgasmPercent', 0);
            updateProgress('sensitivityProgress', 'sensitivityPercent', 0);
        }
    </script>
